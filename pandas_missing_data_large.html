import pandas as pd
import numpy as np
from datetime import datetime, timedelta
import warnings; warnings.filterwarnings("ignore")

np.random.seed(42)

# ── Helper: inject random NaN at a given rate ────────────────────
def inject_nulls(series: pd.Series, rate: float) -> pd.Series:
    """Set ~rate fraction of values to NaN."""
    mask = np.random.rand(len(series)) < rate
    return series.where(~mask, other=np.nan)

# ── Helper: quick audit ──────────────────────────────────────────
def audit(df: pd.DataFrame, label: str = "") -> pd.DataFrame:
    result = pd.DataFrame({
        "missing": df.isnull().sum(),
        "pct":     (df.isnull().sum() / len(df) * 100).round(2),
        "dtype":   df.dtypes,
    })
    nonzero = result[result["missing"] > 0]
    print(f"\n{'─'*50}\n📊 {label} — {len(df):,} rows × {len(df.columns)} cols")
    print(f"   Total missing: {df.isnull().sum().sum():,}"
          f"  ({df.isnull().sum().sum()/df.size*100:.1f}% of all cells)")
    print(nonzero.to_string())
    return nonzero

Dataset 1 — NYC Taxi Trips (1.2 M rows)

Simulates the TLC Yellow Taxi dataset. Missing data comes from GPS failures, meter glitches, and manual entry errors.

Generate & audit the dataset

1,200,000 rows
🏙️ Scenario: A data engineering team receives daily taxi trip dumps. ~8–15% of fields are corrupted — GPS coordinates lost mid-trip, fare amounts blank when meter reboots, passenger count missing for shared rides.
N = 1_200_000

taxi = pd.DataFrame({
    "trip_id":        np.arange(1, N + 1),
    "pickup_dt":      pd.date_range("2024-01-01", periods=N, freq="2s"),
    "pickup_lat":     np.random.uniform(40.50, 40.90, N),
    "pickup_lon":     np.random.uniform(-74.05, -73.75, N),
    "dropoff_lat":    np.random.uniform(40.50, 40.90, N),
    "dropoff_lon":    np.random.uniform(-74.05, -73.75, N),
    "passengers":     np.random.randint(1, 5, N).astype(float),
    "trip_miles":     np.random.exponential(3.5, N),
    "fare_amount":    np.random.exponential(18, N),
    "tip_amount":     np.random.exponential(3, N),
    "payment_type":   np.random.choice(["card", "cash", "app"], N),
    "vendor_id":      np.random.choice([1, 2], N),
})

# Realistic null patterns
taxi["pickup_lat"]  = inject_nulls(taxi["pickup_lat"],  0.08)  # GPS loss
taxi["pickup_lon"]  = taxi["pickup_lat"].isna().map(           # correlated — if lat missing, lon too
    {True: np.nan, False: taxi["pickup_lon"]})
taxi["dropoff_lat"] = inject_nulls(taxi["dropoff_lat"], 0.05)
taxi["fare_amount"] = inject_nulls(taxi["fare_amount"], 0.12)  # meter reboot
taxi["passengers"] = inject_nulls(taxi["passengers"], 0.07)
taxi["tip_amount"]  = taxi["payment_type"].map(               # tip always null for cash
    lambda p: np.nan if p == "cash" else taxi.loc[taxi["payment_type"] == p,
                                                  "tip_amount"].values[0])

Strategy 1 — structured null for tip (cash always null)

# Tip is structurally null for cash — fill with 0, not mean
taxi["tip_amount"] = np.where(
    taxi["payment_type"] == "cash",
    0.0,
    taxi["tip_amount"].fillna(taxi["tip_amount"].median())
)
print(taxi["tip_amount"].isnull().sum())  # → 0
💡 Key insight: Tip is structurally missing for cash transactions — filling with the mean tip would artificially inflate revenue metrics. Always distinguish MCAR (Missing Completely At Random) from MNAR (Missing Not At Random).

Strategy 2 — GPS coordinates (correlated pair)

# Drop rows where pickup GPS is entirely gone — can't map the trip
before = len(taxi)
taxi = taxi.dropna(subset=["pickup_lat", "pickup_lon"])
print(f"Dropped {before - len(taxi):,} rows ({(before-len(taxi))/before*100:.1f}%) with missing pickup GPS")

# Dropoff GPS missing ~5%: forward-fill by trip_id neighbourhood
# (near-consecutive trips share the same zone)
taxi = taxi.sort_values("pickup_dt").reset_index(drop=True)
taxi[["dropoff_lat","dropoff_lon"]] = (
    taxi[["dropoff_lat","dropoff_lon"]]
      .ffill(limit=2)
      .fillna(method="bfill", limit=2)

print(taxi[["dropoff_lat","dropoff_lon"]].isnull().sum())

Strategy 3 — fare_amount via regression on trip_miles

from sklearn.linear_model import LinearRegression

# Fare ≈ base_rate + per_mile × miles  →  use trips where both are known
known   = taxi[taxi["fare_amount"].notna()]
missing = taxi[taxi["fare_amount"].isna()]

model = LinearRegression()
model.fit(known[["trip_miles"]], known["fare_amount"])

taxi.loc[taxi["fare_amount"].isna(), "fare_amount"] = (
    model.predict(missing[["trip_miles"]])
        .clip(2.5)   # minimum NYC fare
)

print(f"Fare R²  = {model.score(known[['trip_miles']], known['fare_amount']):.3f}")
print(f"Missing after imputation: {taxi['fare_amount'].isna().sum()}")

# Passenger count: fill with mode (1 person most common)
taxi["passengers"] = taxi["passengers"].fillna(
    taxi["passengers"].mode()[0]
)

Dataset 2 — Weather Station Network (800 K rows)

Hourly readings from 200 stations across 400 days. Equipment outages, network drops, and sensor calibration gaps create irregular missing blocks.

Generate & characterise

800,000 rows
🌡️ Scenario: A national meteorological service ingests hourly data. Some stations go offline for hours (random outage); others fail systematically overnight (power-saving mode drops humidity sensors).
N_STATIONS = 200;  N_HOURS = 4_000   # 800,000 rows

dates   = pd.date_range("2023-01-01", periods=N_HOURS, freq="h")
station_ids = np.repeat(np.arange(1, N_STATIONS+1), N_HOURS)
times       = np.tile(dates, N_STATIONS)

# Synthetic climate signal + station-level offset
t_base = np.tile(15 + 10*np.sin(np.linspace(0, 4*np.pi, N_HOURS)), N_STATIONS)
t_offset = np.repeat(np.random.uniform(-3, 3, N_STATIONS), N_HOURS)

weather = pd.DataFrame({
    "station_id": station_ids,
    "timestamp":  times,
    "temp_c":     t_base + t_offset + np.random.normal(0, 0.5, N_STATIONS*N_HOURS),
    "humidity":   np.clip(
                     60 + 20*np.sin(np.linspace(0, 8*np.pi, N_STATIONS*N_HOURS))
                     + np.random.normal(0, 3, N_STATIONS*N_HOURS), 10, 100),
    "pressure_hpa": np.random.normal(1013, 8, N_STATIONS*N_HOURS),
    "wind_kph":   np.abs(np.random.normal(18, 10, N_STATIONS*N_HOURS)),
    "precip_mm":  np.abs(np.random.exponential(0.4, N_STATIONS*N_HOURS)),
})

# Station random outage blocks (5–20 consecutive hours each)
for sid in np.random.choice(range(N_STATIONS), 60, replace=False):
    start = np.random.randint(0, N_HOURS - 20)
    length = np.random.randint(5, 20)
    mask = (weather["station_id"] == sid+1) & \
           weather["timestamp"].isin(dates[start:start+length])
    weather.loc[mask, ["temp_c","humidity","pressure_hpa","wind_kph","precip_mm"]] = np.nan

# Systematic: humidity sensors off for all stations 02:00–04:00 UTC
night_mask = weather["timestamp"].dt.hour.isin([2, 3])
weather.loc[night_mask, "humidity"] = np.nan

Strategy — time-aware interpolation per station

weather = weather.sort_values(["station_id", "timestamp"])

# Random outages (short blocks): cubic spline per station per variable
cols_interp = ["temp_c", "pressure_hpa", "wind_kph"]

weather[cols_interp] = (
    weather
    .groupby("station_id")[cols_interp]
    .transform(lambda s: s.interpolate(
        method="cubicspline",
        limit=18,                  # don't interpolate gaps > 18 h
        limit_direction="both"
    ))
)

# Precipitation: exponential backoff — not meaningful to interpolate rain
# Use forward fill with limit=1, then fill remaining with daily median
weather["precip_mm"] = (
    weather
    .groupby("station_id")["precip_mm"]
    .transform(lambda s: s.ffill(limit=1).fillna(s.median()))
)

# Systematic humidity gap: fill using the SAME hour from adjacent days
# (2-day rolling mean of that hour for same station)
def fill_humidity(group):
    group = group.set_index("timestamp")
    group["humidity"] = group["humidity"].interpolate(
        method="time", limit=4, limit_direction="both"
    )
    return group.reset_index()

weather = (
    weather.groupby("station_id", group_keys=False)
           .apply(fill_humidity)
)

print(weather[cols_interp + ["humidity", "precip_mm"]].isnull().sum())

⚠️ Performance note: groupby().apply() on 800K rows with 200 groups is slow in pandas. Use groupby().transform() with vectorised methods whenever possible — it is 5–10× faster. Reserve apply() only when per-group logic cannot be vectorised (like setting a DatetimeIndex inside the group).

Dataset 3 — Stock Market Tick Data (500 K rows)

Minute-level OHLCV data for 50 tickers across 2 years. Trading halts and thin markets create NaN blocks during extended hours.

Generate & audit

500,000 rows
💹 Scenario: A quant fund ingests tick data for 50 tickers. Missing OHLCV rows appear during circuit-breaker halts, after-hours illiquidity, and feed encoding bugs (volume=NaN).
N_TICKERS = 50;  N_MINS = 10_000   # 500,000 rows

tickers  = [f"TKR{i:02d}" for i in range(1, N_TICKERS+1)]
mins     = pd.date_range("2022-01-03 09:30", periods=N_MINS, freq="min")

frames = []
for tkr in tickers:
    price = 100 * np.exp(np.cumsum(np.random.normal(0, 0.001, N_MINS)))
    frames.append(pd.DataFrame({
        "ticker": tkr,
        "ts":     mins,
        "open":   price * (1 + np.random.uniform(-0.002, 0.002, N_MINS)),
        "high":   price * (1 + np.random.uniform(0, 0.005, N_MINS)),
        "low":    price * (1 - np.random.uniform(0, 0.005, N_MINS)),
        "close":  price,
        "volume": np.abs(np.random.normal(50_000, 15_000, N_MINS)),
    }))

stocks = pd.concat(frames, ignore_index=True)

# Halt: 3 random 30-min blocks per 10 tickers = silent prices
for tkr in np.random.choice(tickers, 10, replace=False):
    for _ in range(3):
        start = np.random.randint(0, N_MINS - 30)
        halt_ts = mins[start:start+30]
        m = (stocks["ticker"] == tkr) & stocks["ts"].isin(halt_ts)
        stocks.loc[m, ["open","high","low","close","volume"]] = np.nan

# Encoding bug: volume randomly null
stocks["volume"] = inject_nulls(stocks["volume"], 0.04)

audit(stocks, "Stocks")

Strategy — last-observation-carried-forward (LOCF) + VWAP-based volume fill

stocks = stocks.sort_values(["ticker", "ts"]).reset_index(drop=True)

# ── OHLC during halts → carry last known price (LOCF) ────────────
price_cols = ["open", "high", "low", "close"]
stocks[price_cols] = (
    stocks.groupby("ticker")[price_cols]
          .transform(lambda s: s.ffill(limit=30))  # max 30-min carry
)

# During halts, set volume = 0 (no trades happened)
halt_mask = (
    stocks.groupby("ticker")["close"]
          .transform(lambda s: s.isna())       # rows still NaN after LOCF = halt
)
stocks.loc[halt_mask, "volume"] = 0.0
stocks[price_cols] = stocks[price_cols].fillna(method="bfill")

# ── Random volume nulls → fill with 15-min rolling median per ticker ─
stocks["volume"] = stocks.groupby("ticker")["volume"].transform(
    lambda s: s.fillna(s.rolling(15, min_periods=1, center=True).median())
)

print(stocks.isnull().sum().sum())
Output
0   ← all missing resolved
✅ Finance rule: Price LOCF is the industry standard during trading halts — the last trade is the best available price. Volume = 0 is correct (no shares changed hands). Never interpolate prices through a halt as that implies trades occurred when none did.

Strategy — diagnosis-aware imputation + KNN for BMI

KNN / MICE
from sklearn.impute import KNNImputer
from sklearn.preprocessing import LabelEncoder

# Step 1: MNAR labs — fill with "not applicable" sentinel value
# (do NOT use mean — a cardiac patient does NOT have a hidden HbA1c)
ehr["hba1c_available"]   = ehr["hba1c"].notna().astype("int8")
ehr["troponin_available"] = ehr["troponin"].notna().astype("int8")

ehr["hba1c"]   = ehr["hba1c"].fillna(-1)      # sentinel: not diabetic
ehr["troponin"] = ehr["troponin"].fillna(-1)   # sentinel: not cardiac

# Step 2: Age — median imputation (3% missing, MAR)
ehr["age"] = ehr["age"].fillna(ehr["age"].median())

# Step 3: Gender — mode within diagnosis group
ehr["gender"] = ehr.groupby("diagnosis")["gender"].transform(
    lambda s: s.fillna(s.mode()[0] if not s.mode().empty else "U")
)

# Step 4: BMI — KNN on age, glucose, creatinine (correlated features)
le = LabelEncoder()
ehr["diag_enc"] = le.fit_transform(ehr["diagnosis"])

knn_feats = ["age", "glucose", "creatinine", "diag_enc", "bmi"]
knn_data  = ehr[knn_feats].copy()

# Use chunked KNN for large datasets (avoid 300K×300K distance matrix)
CHUNK = 50_000
imputer = KNNImputer(n_neighbors=7)

# Fit on complete rows only
complete_rows = knn_data.dropna()
imputer.fit(complete_rows)

# Transform all rows in chunks
parts = []
for i in range(0, len(knn_data), CHUNK):
    parts.append(
        pd.DataFrame(imputer.transform(knn_data.iloc[i:i+CHUNK]),
                     columns=knn_feats)
    )
ehr[knn_feats] = pd.concat(parts).values

# Clean up encoder column
ehr.drop(columns=["diag_enc"], inplace=True)

print(ehr.isnull().sum())

    ecom.dropna(subset=["ship_days"])

)
ecom = ecom.join(median_lookup, on=["country", "category"])
ecom["ship_days"] = ecom["ship_days"].fillna(ecom["med_ship"])
ecom.drop(columns=["med_ship"], inplace=True)

# ── Compute final order value ────────────────────────────────────
ecom["order_value"] = (
    ecom["item_price"] * ecom["quantity"]
    * (1 - ecom["discount_pct"] / 100)
)

Dataset 6 — Industrial IoT Sensors (5 M rows)

10-second readings from 250 factory floor sensors over 8 months. Network packet loss and sensor saturation create both sparse and block-level NaN.

Efficient chunked processing for 5 M rows

5,000,000 rowsMemory efficient
🏭 Scenario: A predictive maintenance system must impute missing sensor readings before feeding a neural-network anomaly detector. The 5M-row dataset does not fit in memory twice — use chunking and Categorical dtype to minimise RAM.
# ── Generate (in practice you'd read from Parquet / HDF5) ───────
N = 5_000_000
N_SENSORS = 250
READINGS_PER = N // N_SENSORS          # 20,000 per sensor

sensor_ids = np.repeat(np.arange(1, N_SENSORS+1), READINGS_PER)
timestamps  = np.tile(pd.date_range("2023-06-01", periods=READINGS_PER,
                                    freq="10s"), N_SENSORS)

iot = pd.DataFrame({
    "sensor_id": pd.Categorical(sensor_ids),  # Categorical saves ~36 MB RAM
    "ts":        timestamps,
    "temp":      np.random.normal(75, 5, N),
    "vibration": np.abs(np.random.normal(0.5, 0.1, N)),
    "pressure":  np.random.normal(200, 15, N),
    "current_a": np.random.normal(12, 2, N),
})

Strategy — vectorised per-sensor linear interpolation at scale

Fast path
# Sort once — critical for ffill/interpolate correctness
iot = iot.sort_values(["sensor_id", "ts"]).reset_index(drop=True)

sensor_cols = ["temp", "vibration", "pressure", "current_a"]

# ── Fast path: transform with linear interpolate (limit=60 = 10 min) ─
# transform() is vectorised and does NOT construct per-group DataFrames
iot[sensor_cols] = (
    iot.groupby("sensor_id")[sensor_cols]
       .transform(lambda s:
           s.interpolate(method="linear", limit=60, limit_direction="both")
       )
)

# ── Any remaining NaN (start/end of series beyond limit) → sensor median
sensor_medians = iot.groupby("sensor_id")[sensor_cols].transform("median")
iot[sensor_cols] = iot[sensor_cols].fillna(sensor_medians)

print(f"Remaining nulls: {iot[sensor_cols].isnull().sum().sum()}")

# ── Add rolling stats for anomaly detection features ────────────
iot["temp_roll5m"] = (
    iot.groupby("sensor_id")["temp"]
       .transform(lambda s:
           s.rolling(30, min_periods=5).mean()   # 30 × 10s = 5-min window
       )
)
# Rolling produces NaN at window start → back-fill with first valid
iot["temp_roll5m"] = (
    iot.groupby("sensor_id")["temp_roll5m"]
       .transform(lambda s: s.bfill())
)

print(iot.isnull().sum())

Production ML Pipeline — sklearn + pandas

A reusable, cross-validated imputation pipeline that prevents data leakage — fit only on training data, transform both train and test identically.

Full leakage-free pipeline with ColumnTransformer

sklearn Pipeline
⛔ Common mistake: Calling df.fillna(df.mean()) on the whole dataset before train/test split leaks test-set statistics into training — inflating cross-validation scores.
from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer
from sklearn.impute import SimpleImputer, KNNImputer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.ensemble import GradientBoostingClassifier
from sklearn.model_selection import cross_val_score
from sklearn.pipeline import make_pipeline

# ── Use the EHR dataset as example ──────────────────────────────
# Reload raw (with nulls)
ehr_raw = ehr.copy()   # assume 'ehr' still has original NaN structure

NUM_FEATS = ["age", "los_days", "glucose", "creatinine", "bmi"]
CAT_FEATS = ["gender", "diagnosis"]
TARGET    = "readmitted_30d"

numeric_transformer = Pipeline([
    ("imputer", KNNImputer(n_neighbors=5)),     # fit on train only ✅
    ("scaler",  StandardScaler()),
])

categorical_transformer = Pipeline([
    ("imputer", SimpleImputer(strategy="most_frequent")),
    ("encoder", OneHotEncoder(handle_unknown="ignore", sparse_output=False)),
])

preprocessor = ColumnTransformer(transformers=[
    ("num", numeric_transformer,    NUM_FEATS),
    ("cat", categorical_transformer, CAT_FEATS),
])

clf_pipeline = Pipeline([
    ("prep",  preprocessor),
    ("model", GradientBoostingClassifier(n_estimators=100, random_state=42)),
])

X = ehr_raw[NUM_FEATS + CAT_FEATS]
y = ehr_raw[TARGET]

# 5-fold CV — imputer sees only TRAINING fold each time
scores = cross_val_score(
    clf_pipeline, X, y,
    cv=5, scoring="roc_auc", n_jobs=-1,
    verbose=1
)
print(f"ROC-AUC: {scores.mean():.3f} ± {scores.std():.3f}")
Output
ROC-AUC: 0.714 ± 0.008    ← no leakage, honest estimate
✅ Best practice: Wrapping imputation inside a Pipeline and passing it to cross_val_score or GridSearchCV guarantees the imputer is re-fit on each training fold — identical to what happens at production serving time.

# ── 1. Use nullable integer dtype to avoid float64 for int+NaN ──
#    Standard int64 cannot hold NaN → pandas silently upcasts to float64
df["order_count"] = df["order_count"].astype("Int64")   # capital I
print(df["order_count"].dtype)         # → Int64 (nullable)

# ── 2. Categorical for low-cardinality string columns ────────────
df["category"] = df["category"].astype("category")
# Before: object (pointer per row) — ~64 B × 2M = 128 MB
# After:  category (int8 code)     — ~1 B  × 2M = 2 MB  ✅

# ── 3. Downcast after filling ─────────────────────────────────
df["age"] = pd.to_numeric(df["age"].fillna(0), downcast="integer")

Chunked CSV processing — when dataset > RAM

import pandas as pd

CHUNK_SIZE = 500_000
results    = []

print("Processing large CSV in chunks...")

with pd.read_csv("large_iot.csv", chunksize=CHUNK_SIZE,
                 parse_dates=["ts"],
                 dtype={"sensor_id": "int16",
                        "temp":      "float32"}) as reader:

    for i, chunk in enumerate(reader):
        # Per-chunk: ffill within sensor then fill with chunk median
        chunk = chunk.sort_values(["sensor_id", "ts"])
        chunk["temp"] = (
            chunk.groupby("sensor_id")["temp"]
                 .transform(lambda s: s.ffill(limit=6)
                              .fillna(s.median()))
        )
        results.append(chunk)
        print(f"  chunk {i+1}: {len(chunk):,} rows processed", end="\r")

df_cleaned = pd.concat(results, ignore_index=True)
print(f"\nDone — {len(df_cleaned):,} rows cleaned")
⚠️ Chunk edge case: ffill across chunk boundaries will miss values that span two chunks. For strict continuity, either sort the full file first (e.g. via dask), or use a stateful boundary-carry approach where you save the last valid row of each sensor from the prior chunk.

MCAR — Missing Completely At Random: no relationship between missingness and any other variable. Safe to impute with mean/median.
MAR — Missing At Random: missingness depends on observed variables (e.g. age missing more often for elderly). Group-aware imputation handles this.
MNAR — Missing Not At Random: missingness depends on the hidden value itself (e.g. high-income people less likely to report income). Flag + sentinel, never mean-fill.
Golden rule: Always create a _was_missing indicator before imputing anything for ML. The fact of missingness is often the strongest signal.
