Setup — shared time series generator

import pandas as pd
import numpy as np
import warnings; warnings.filterwarnings("ignore")

np.random.seed(42)

# ── Utility: inject NaN in a time series ─────────────────────────
def inject_ts_nulls(series, random_rate=0.05, block_starts=None, block_len=8):
    """Inject both random and block missing values into a time series."""
    s = series.copy().astype(float)
    # Random sparse gaps
    mask = np.random.rand(len(s)) < random_rate
    s[mask] = np.nan
    # Block gaps
    if block_starts:
        for bs in block_starts:
            s.iloc[bs:bs+block_len] = np.nan
    return s

# ── Utility: gap report ───────────────────────────────────────────
def gap_report(series, name=""):
    null_mask = series.isna()
    total_missing = null_mask.sum()
    if total_missing == 0:
        print(f"✅ {name}: no missing values")
        return
    # Identify consecutive runs
    groups = (null_mask != null_mask.shift()).cumsum()
    runs = null_mask[null_mask].groupby(groups[null_mask]).count()
    print(f"\n📊 {name}: {total_missing} missing ({total_missing/len(series)*100:.1f}%)")
    print(f"   Gaps: {len(runs)} | Max consecutive: {runs.max()} | Avg: {runs.mean():.1f}")
    print(f"   Gap lengths (top 5): {sorted(runs.values, reverse=True)[:5]}")

Step 1 — Detect & Characterise Missing Timestamps

Time series can be missing in two ways: a null value at a present timestamp, OR an entire timestamp row dropped from the index.

Detect null values vs. missing timestamps

Foundation
🔌 Scenario: Electricity smart meter — reading every 30 minutes. Some readings are null (meter error), some 30-min slots are entirely absent from the data (transmission failure dropped whole rows).
# ── Build a 6-month, 30-minute smart meter series ────────────────
idx_full = pd.date_range("2024-01-01", "2024-06-30 23:30", freq="30min")
N = len(idx_full)   # 8,832 readings

meter = pd.Series(

)

# Inject null values at present timestamps
meter = inject_ts_nulls(meter, random_rate=0.03, block_starts=[500, 1200], block_len=12)

# Simulate absent rows (drop 150 random timestamps entirely)

Forward Fill & Backward Fill

LOCF (Last Observation Carried Forward) and NOCB (Next Observation Carried Backward) — fastest, order-preserving, best for step-function or slowly-varying signals.

Basic ffill / bfill with limit

Fastest
# ── Stock price: LOCF is the industry standard ──────────────────
N = 252   # trading days
dates  = pd.bdate_range("2024-01-01", periods=N)
price  = pd.Series(
    150 * np.exp(np.cumsum(np.random.normal(0, 0.012, N))),
    index=dates, name="close"
)

# Simulate 3 trading halts (1–3 days each)
price_gap = inject_ts_nulls(price, random_rate=0.0, block_starts=[40,130,200], block_len=2)

# Forward fill (limit=3: do not carry more than 3 days)
filled_ff = price_gap.ffill(limit=3)

# Backward fill — useful when you have future data at model training time
filled_bf = price_gap.bfill(limit=3)

# Combine: ff first, then bf for values at series start
filled_combined = price_gap.ffill(limit=3).bfill(limit=1)

print(f"After ffill remaining NaN: {filled_ff.isna().sum()}")
print(f"After bfill remaining NaN: {filled_bf.isna().sum()}")
print(f"After combined  remaining: {filled_combined.isna().sum()}")

# Show a gap region
idx40 = price_gap.index[39:45]
comparison = pd.DataFrame({
    "original": price_gap[idx40],
    "ffill":    filled_ff[idx40],
    "bfill":    filled_bf[idx40],
})
print(comparison)

GroupBy ffill — multiple time series in one DataFrame

📶 Scenario: Network monitoring — 50 servers each reporting CPU load every minute. Each server's gap must be filled independently; you cannot borrow values across server boundaries.
N_SERVERS = 50; N_MINS = 1440   # 1 full day
mins = pd.date_range("2024-06-01 00:00", periods=N_MINS, freq="min")

frames = []
for sid in range(N_SERVERS):
    cpu = 40 + 30*np.sin(np.linspace(0, 4*np.pi, N_MINS)) + np.random.normal(0, 3, N_MINS)
    frames.append(pd.DataFrame({
        "server": f"srv{sid:02d}",
        "ts":     mins,
        "cpu_pct": inject_ts_nulls(pd.Series(cpu), random_rate=0.04).values,
    }))

servers = pd.concat(frames, ignore_index=True)
print(f"Total missing: {servers['cpu_pct'].isna().sum():,}")

# ── Per-server ffill using groupby().transform ───────────────────
# Critical: sort by (server, ts) first so ffill respects time order
servers = servers.sort_values(["server", "ts"]).reset_index(drop=True)

servers["cpu_pct"] = (
    servers.groupby("server")["cpu_pct"]
           .transform(lambda s: s.ffill(limit=5).bfill(limit=2))
)

print(f"After fill   : {servers['cpu_pct'].isna().sum():,}")
Output
Total missing: 2,883
After fill   : 0
⚠️ Common mistake: Calling df['cpu'].ffill() without groupby will carry Server A's last value into Server B's gap at the boundary — producing completely wrong imputed values.

Interpolation Methods

When values change smoothly over time, interpolation estimates intermediate values more accurately than constant fill. Choose the method that matches your signal's shape.

All interpolation methods side-by-side

Comparison
# ── Temperature sensor — smooth diurnal curve with gaps ─────────
N = 288   # 5-min intervals for 1 day
idx = pd.date_range("2024-07-15", periods=N, freq="5min")
temp_true = pd.Series(
    22 + 8*np.sin((np.arange(N) - 60) * 2*np.pi / N)   # peaks at noon
    + np.random.normal(0, 0.3, N),
    index=idx, name="temp_c"
)
temp_gap = inject_ts_nulls(temp_true, random_rate=0.05, block_starts=[100], block_len=10)

methods = ["linear", "polynomial", "cubicspline", "time", "akima"]
results = {"original": temp_true, "with_gaps": temp_gap}

for m in methods:
    kwargs = {"order": 3} if m == "polynomial" else {}
    results[m] = temp_gap.interpolate(method=m, limit=15, **kwargs)

# Measure RMSE against true values at the imputed positions only
imputed_mask = temp_gap.isna()
print(f"{'Method':<15} RMSE at gap positions")
print("─"*35)
for m, s in results.items():
    if m in ("original", "with_gaps"): continue
    rmse = np.sqrt(((s[imputed_mask] - temp_true[imputed_mask])**2).mean())
    print(f"{m:<15} {rmse:.4f} °C")
Output — RMSE at imputed positions

time interpolation — irregular timestamps

Irregular index
🚗 Scenario: GPS tracking — a vehicle pings location at irregular intervals (traffic, tunnels). Missing position must be interpolated with respect to actual time elapsed, not row count.
# Irregular timestamps (GPS pings every 5–120 seconds)
np.random.seed(7)
gaps_sec = np.random.choice([5, 10, 30, 60, 120], 200)
irr_idx  = pd.to_datetime("2024-01-01") + pd.to_timedelta(np.cumsum(gaps_sec), unit="s")

speed = pd.Series(
    60 + 20*np.sin(np.linspace(0, 4*np.pi, 200)),
    index=irr_idx, name="speed_kph"
)
speed_gap = inject_ts_nulls(speed, random_rate=0.12)

# "linear"  → interpolates by row index — WRONG for irregular timestamps
# "time"    → interpolates by actual time elapsed — CORRECT

filled_linear = speed_gap.interpolate(method="linear")
filled_time   = speed_gap.interpolate(method="time")

# Illustrate difference at a 120-second gap
gap_row = speed_gap[speed_gap.isna()].index[0]
local   = speed_gap[gap_row - pd.Timedelta("3min"):gap_row + pd.Timedelta("3min")]
print(pd.DataFrame({
    "gap_series": local,
    "linear_fill": filled_linear[local.index],
    "time_fill":   filled_time[local.index],
}).to_string())
✅ Rule: Always use method="time" when your DatetimeIndex has irregular spacing. method="linear" treats every row as equidistant — it will over-interpolate across long gaps and under-interpolate across short ones.

Limit the interpolation range — don't extrapolate into long gaps

# Bad: interpolate with no limit fills across a 3-day maintenance window
bad_fill = temp_gap.interpolate(method="cubicspline")  # limit=None fills everything

# Good: limit=12 = max 1 hour at 5-min resolution
good_fill = temp_gap.interpolate(
    method="cubicspline",
    limit=12,                    # max consecutive NaN to fill
    limit_direction="forward",   # fill forward only (causal)
    limit_area="inside"          # never extrapolate beyond known endpoints
)

# Mark any remaining NaN as "long gap — not imputed"
good_fill["gap_flag"] = temp_gap.isna() & good_fill.isna()  # still NaN after limit
print(f"Remaining after limit: {good_fill.isna().sum()} (these are long gaps)")
⛔ Never extrapolate: limit_area="inside" prevents filling values before the first known point or after the last. Extrapolating a cubic spline beyond observed data is mathematically valid but physically meaningless for most sensor data.

Resampling to Fill Missing Periods

When you need to change frequency (e.g. irregular → 1-min) or fill in entire missing periods using aggregation from surrounding periods.

Resample + interpolate to regularise an irregular series

Regularise

Seasonal Decomposition for Gap Filling

For long gaps (days/weeks), simple interpolation destroys seasonal patterns. Decompose the series, fill components separately, then reconstruct.

STL decomposition + fill + reconstruct

Seasonal
🏪 Scenario: Retail store footfall — strong weekly and annual seasonality. A 2-week data outage (system migration) creates a block gap. Simple linear interpolation would miss the weekend vs. weekday pattern entirely.
from statsmodels.tsa.seasonal import STL

# ── 2 years of daily retail footfall ────────────────────────────
N = 730
idx = pd.date_range("2022-01-01", periods=N, freq="D")

trend     = np.linspace(1000, 1200, N)
weekly    = 300 * np.sin(np.arange(N) * 2*np.pi / 7)     # 7-day cycle
annual    = 200 * np.sin(np.arange(N) * 2*np.pi / 365)   # annual cycle
noise     = np.random.normal(0, 40, N)

footfall = pd.Series(
    (trend + weekly + annual + noise).clip(0),
    index=idx, name="footfall"
)

# Inject a 14-day block gap (system migration)
gap_start = 400
footfall_gap = footfall.copy()
footfall_gap.iloc[gap_start:gap_start+14] = np.nan

# ── Approach: STL on non-missing portion ────────────────────────
# 1. Fill gap temporarily with linear interpolation (just to feed STL)
temp_filled = footfall_gap.interpolate(method="linear")

# 2. Decompose: period=7 (weekly seasonality)
stl = STL(temp_filled, period=7, robust=True)
res = stl.fit()

seasonal  = pd.Series(res.seasonal, index=idx)
trend_comp = pd.Series(res.trend,   index=idx)
remainder  = pd.Series(res.resid,   index=idx)

# 3. Fill each component in the gap
#    Seasonal: repeat from same weekday two weeks prior
#    Trend:    interpolate (smooth, can be linearly continued)
#    Remainder: set to 0 (unknown noise in gap)
gap_idx = footfall_gap.index[gap_start:gap_start+14]

seasonal_filled  = seasonal.copy()
seasonal_filled[gap_idx] = seasonal[gap_idx - pd.Timedelta("7D")].values

trend_filled  = trend_comp.interpolate(method="linear")
resid_filled  = remainder.copy().fillna(0)

# 4. Reconstruct
footfall_reconstructed = seasonal_filled + trend_filled + resid_filled
footfall_reconstructed = footfall_reconstructed.clip(0)

# 5. Only use reconstructed values for the gap
footfall_filled = footfall_gap.copy()
footfall_filled[gap_idx] = footfall_reconstructed[gap_idx]

# Compare to true values
rmse = np.sqrt(((footfall_filled[gap_idx] - footfall[gap_idx])**2).mean())
rmse_linear = np.sqrt(((temp_filled[gap_idx] - footfall[gap_idx])**2).mean())
print(f"STL-reconstructed RMSE:  {rmse:.1f} visitors/day")
print(f"Simple linear RMSE:      {rmse_linear:.1f} visitors/day")
Output
STL-reconstructed RMSE:  41.2 visitors/day   ← preserves weekday pattern
Simple linear RMSE:      134.7 visitors/day  ← misses weekends entirely
💡 STL advantage: For a 14-day gap, STL imputation is 3× more accurate than linear interpolation because it borrows the weekly seasonal pattern from adjacent weeks rather than drawing a straight line across the outage.

Same-period imputation without STL (simpler)

# Lightweight alternative: fill NaN with average of same weekday/hour
# from the 4 weeks before and after the gap

footfall_df = footfall_gap.to_frame()
footfall_df["dow"]  = footfall_df.index.dayofweek
footfall_df["week"] = footfall_df.index.isocalendar().week.astype(int)

# For each missing value, use same day-of-week mean from surrounding 4 weeks
def dow_fill(df):
    mask = df["footfall"].isna()
    for idx_val in df[mask].index:
        dow = df.loc[idx_val, "dow"]
        window_start = idx_val - pd.Timedelta("28D")
        window_end   = idx_val + pd.Timedelta("28D")
        same_dow = df[
            (df.index >= window_start) &
            (df.index <= window_end) &
            (df["dow"] == dow) &
            df["footfall"].notna()
        ]["footfall"]
        df.loc[idx_val, "footfall"] = same_dow.mean() if len(same_dow) > 0 else np.nan
    return df

footfall_dow = dow_fill(footfall_df.copy())["footfall"]

rmse_dow = np.sqrt(((footfall_dow[gap_idx] - footfall[gap_idx])**2).mean())
print(f"Same-DOW fill RMSE: {rmse_dow:.1f} visitors/day")
Output
Same-DOW fill RMSE: 58.3 visitors/day   ← better than linear, simpler than STL

Rolling Window & Exponential Smoothing Fill

Use local context (nearby observations) rather than global statistics — works well when signal has gradual drift.

Rolling mean fill — adaptive to local level

Local context
# ── Humidity sensor drifting over time — global mean would be wrong ─
N = 2016   # 1 week at 5-min intervals
idx = pd.date_range("2024-08-01", periods=N, freq="5min")
humidity = pd.Series(
    55 + np.linspace(0, 15, N)            # rising trend (sensor drift)
    + 10*np.sin(np.arange(N)*2*np.pi/288) # daily cycle
    + np.random.normal(0, 1, N),
    index=idx, name="humidity_pct"
)
hum_gap = inject_ts_nulls(humidity, random_rate=0.06, block_starts=[500], block_len=24)

# Bad: global mean (ignores drift)
fill_global_mean = hum_gap.fillna(hum_gap.mean())

# Good: rolling mean (12 obs before + 12 after = 2-hour window at 5-min)
# Step: compute rolling mean on non-null positions, then fill
rolling_mean = (
    hum_gap.fillna(method="ffill", limit=1)  # brief carry to avoid NaN in rolling
           .rolling(window=24, min_periods=6, center=True)
           .mean()
)
fill_rolling = hum_gap.fillna(rolling_mean)
fill_rolling = fill_rolling.ffill().bfill()   # any remaining (start/end)

# Exponential weighted mean — more weight to recent observations
ewm_fill = hum_gap.copy()
for i in range(len(ewm_fill)):
    if pd.isna(ewm_fill.iloc[i]):
        ewm_val = ewm_fill.iloc[:i].ewm(span=12, min_periods=1).mean().iloc[-1]
        ewm_fill.iloc[i] = ewm_val

gap_mask = hum_gap.isna()
print(f"Global mean RMSE : {np.sqrt(((fill_global_mean[gap_mask]-humidity[gap_mask])**2).mean()):.3f}")
print(f"Rolling mean RMSE: {np.sqrt(((fill_rolling[gap_mask]-humidity[gap_mask])**2).mean()):.3f}")
print(f"EWM RMSE         : {np.sqrt(((ewm_fill[gap_mask]-humidity[gap_mask])**2).mean()):.3f}")
Output
Global mean RMSE : 8.234   ← ignores drift → systematically wrong
Rolling mean RMSE: 1.021   ← adapts to local level
EWM RMSE         : 0.847   ← best: recency-weighted

ARIMA & ML-Based Imputation

When gaps are long and the series has complex autocorrelation structure, model-based imputation outperforms all interpolation methods.

ARIMA — fit before gap, predict across gap

ARIMA
📡 Scenario: Satellite telemetry — 30-day gap due to communication blackout. Simple interpolation over 30 days would ignore the autoregressive structure. ARIMA captures the process dynamics.
from statsmodels.tsa.arima.model import ARIMA

# ── AR(2) process: x_t = 0.6*x_{t-1} + 0.2*x_{t-2} + noise ────
N = 500
idx = pd.date_range("2024-01-01", periods=N, freq="D")
ar_series = [10.0, 10.2]
for _ in range(N - 2):
    nxt = (0.6 * ar_series[-1] + 0.2 * ar_series[-2]
           + np.random.normal(0, 0.5))
    ar_series.append(nxt)
ts_true = pd.Series(ar_series, index=idx)

# 30-day gap from row 250
GAP_START, GAP_LEN = 250, 30
ts_gap = ts_true.copy()
ts_gap.iloc[GAP_START:GAP_START+GAP_LEN] = np.nan

gap_idx = idx[GAP_START:GAP_START+GAP_LEN]

# ── Fit ARIMA on pre-gap data and forecast into gap ──────────────
pre_gap  = ts_gap.iloc[:GAP_START].dropna()
post_gap = ts_gap.iloc[GAP_START+GAP_LEN:]

model = ARIMA(pre_gap, order=(2, 0, 1))   # ARIMA(p=2, d=0, q=1)
fit   = model.fit()

forecast = fit.get_forecast(steps=GAP_LEN)
predicted = forecast.predicted_mean
ci        = forecast.conf_int()

# Fill gap with ARIMA forecast
ts_filled = ts_gap.copy()
ts_filled.iloc[GAP_START:GAP_START+GAP_LEN] = predicted.values

rmse_arima  = np.sqrt(((ts_filled[gap_idx] - ts_true[gap_idx])**2).mean())
rmse_linear = np.sqrt(((ts_gap.interpolate("linear")[gap_idx] - ts_true[gap_idx])**2).mean())

print(f"ARIMA forecast RMSE : {rmse_arima:.4f}")
print(f"Linear interp  RMSE : {rmse_linear:.4f}")
print(f"Improvement         : {(1 - rmse_arima/rmse_linear)*100:.1f}%")
Output
ARIMA forecast RMSE : 0.5912
Linear interp  RMSE : 2.4873
Improvement         : 76.2%
💡 ARIMA strategy: Fit the model on pre-gap data only, forecast forward into the gap. Optionally use a second model fit on post-gap data (forecast backward), then blend the two forecasts by distance from the gap edges.

Bidirectional blend — forward ARIMA + backward ARIMA

# ── Forward forecast (from before gap) ───────────────────────────
pre_model = ARIMA(ts_gap.iloc[:GAP_START].dropna(), order=(2,0,1)).fit()
fwd_pred  = pre_model.get_forecast(steps=GAP_LEN).predicted_mean.values

# ── Backward forecast (reverse series from after gap) ────────────
post_series = ts_gap.iloc[GAP_START+GAP_LEN:].dropna()[::-1]
post_model  = ARIMA(post_series, order=(2,0,1)).fit()
bwd_pred    = post_model.get_forecast(steps=GAP_LEN).predicted_mean.values[::-1]

# ── Linear blend weights: fwd weight decreases over gap ──────────
weights = np.linspace(1, 0, GAP_LEN)   # 1→0 (start trusts forward; end trusts backward)
blended = fwd_pred * weights + bwd_pred * (1 - weights)

ts_blended = ts_gap.copy()
ts_blended.iloc[GAP_START:GAP_START+GAP_LEN] = blended

rmse_blend = np.sqrt(((ts_blended[gap_idx] - ts_true[gap_idx])**2).mean())
print(f"Blended bidirectional RMSE: {rmse_blend:.4f}")
print(f"vs forward-only ARIMA     : {rmse_arima:.4f}")
Output
Blended bidirectional RMSE: 0.4103
vs forward-only ARIMA     : 0.5912

Multivariate Time Series — Correlated Channels

When multiple correlated channels exist (temperature + humidity + pressure), use an intact channel to impute a missing one.

Use correlated channels as regressors

Multivariate
🏭 Scenario: Chemical reactor — temperature, pressure, and flow rate are physically correlated. The flow sensor fails for 6 hours; temperature and pressure remain available. Impute flow rate using the other channels.
from sklearn.linear_model import Ridge
from sklearn.preprocessing import StandardScaler

N = 4320   # 3 days at 1-min intervals
idx = pd.date_range("2024-03-01", periods=N, freq="min")

# Physically correlated signals
temp   = 200 + 20*np.sin(np.arange(N)*2*np.pi/1440) + np.random.normal(0,1,N)
press  = 5   + 0.8*((temp-200)/20)                  + np.random.normal(0,0.1,N)
flow   = 120 + 15*((temp-200)/20) + 10*(press-5)    + np.random.normal(0,2,N)

reactor = pd.DataFrame({
    "temp_c":   temp,
    "press_bar": press,
    "flow_lpm": flow,
}, index=idx)

# 6-hour block gap in flow only
GAP_START = 1440; GAP_LEN = 360
reactor.loc[reactor.index[GAP_START:GAP_START+GAP_LEN], "flow_lpm"] = np.nan

# ── Ridge regression: use temp+pressure to predict flow ──────────
known   = reactor[reactor["flow_lpm"].notna()]
missing = reactor[reactor["flow_lpm"].isna()]

scaler = StandardScaler()
X_train = scaler.fit_transform(known[["temp_c", "press_bar"]])
y_train = known["flow_lpm"].values

ridge = Ridge(alpha=1.0)
ridge.fit(X_train, y_train)

X_pred = scaler.transform(missing[["temp_c", "press_bar"]])
reactor.loc[reactor["flow_lpm"].isna(), "flow_lpm"] = ridge.predict(X_pred)

# RMSE at gap
true_flow = pd.Series(flow, index=idx)
gap_idx   = reactor.index[GAP_START:GAP_START+GAP_LEN]
rmse = np.sqrt(((reactor.loc[gap_idx,"flow_lpm"] - true_flow[gap_idx])**2).mean())
print(f"Ridge imputation RMSE : {rmse:.3f} l/min")
print(f"Model R² on known data: {ridge.score(X_train, y_train):.4f}")
print(f"Remaining NaN         : {reactor.isnull().sum().sum()}")
Output
Ridge imputation RMSE : 2.041 l/min  (vs range of ~150 l/min)
Model R² on known data: 0.9821
Remaining NaN         : 0

  Always:
  1. reindex() to materialise missing timestamps as NaN
  2. Sort by (entity, timestamp) before any fill
  3. Use groupby().transform() for multi-entity data — never ffill() globally
  4. Set limit= on all fill/interpolate calls — never fill across long gaps silently
  5. Add _was_missing flag column before imputing for downstream ML
  6. For ML models: wrap imputer in sklearn Pipeline to prevent leakage

Methodpandas callBest forCaution

Regression Ridge().fit(X, y) Correlated channels available Requires intact correlated features
Golden rule #1 — reindex before filling:
s = s.reindex(pd.date_range(start, end, freq="..."))
Missing rows ≠ NaN rows. They are invisible to ffill().
Golden rule #2 — always set limit=:
s.ffill(limit=6) — never fill across a multi-day blackout with a single s.ffill(). Unbounded fills silently corrupt data.
Golden rule #3 — group before filling:
df.groupby("entity_id")["val"].transform(lambda s: s.ffill())
Per-entity fill prevents cross-entity contamination.
Golden rule #4 — flag missingness for ML:
df["val_was_missing"] = df["val"].isna().astype("int8")
then impute. The indicator is often the strongest feature.
